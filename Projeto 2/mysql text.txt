
CREATE TABLE ALUNO (
    ID_ALUNO INT NOT NULL PRIMARY KEY,
    NOME VARCHAR(40),
    ENDERECO VARCHAR(40),
    MATRICULA VARCHAR(40),
    CR INT
);

CREATE TABLE PROFESSOR (
    ID_PROFESSOR INT NOT NULL PRIMARY KEY,
    NOME VARCHAR(40),
    ENDERECO VARCHAR(40),
    SIAPE VARCHAR(40),
    MATRICULA_UFF VARCHAR(40)
);

CREATE TABLE DISCIPLINA (
    ID_DISCIPLINA INT NOT NULL PRIMARY KEY,
    TITULO VARCHAR(40),
    TIPO VARCHAR(40),
    ID_PROFESSOR INTEGER,
    CONSTRAINT FK_PROFESSOR FOREIGN KEY (ID_PROFESSOR) REFERENCES PROFESSOR (ID_PROFESSOR) 
);

CREATE TABLE TURMA (
    ID_TURMA INT NOT NULL PRIMARY KEY,
    ID_DISCIPLINA INT NOT NULL,
    ANO VARCHAR(40),
    NUMERO_DE_VAGAS INT,
    PERIODO VARCHAR(40),
    CONSTRAINT T_DISCIPLINA FOREIGN KEY (ID_DISCIPLINA) REFERENCES DISCIPLINA (ID_DISCIPLINA)
);
